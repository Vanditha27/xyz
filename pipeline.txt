pipeline{
    agent any
    stages{
        stage('hello'){
            steps{
                echo"hi all"
            }
        }
        stage('bye'){
            steps{
                echo"bye all"
            }
        }
        stage('loops'){
            steps{
                script{
                    for(a in 1..10){
                        echo"Number is =${a}"
                    }
                }
            }
        }
    }
}
